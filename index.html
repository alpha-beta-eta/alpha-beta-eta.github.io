Scheme是Lisp的一种方言.

Greatest Common Divisor

(define (gcd a b)
  (if (= b 0)
      a
      (gcd b (remainder a b))))
(define (ext-gcd a b)
  (if (= b 0)
      (values a 1 0)
      (let ((q (quotient a b))
            (r (remainder a b)))
        (let-values (((d m n) (ext-gcd b r)))
          (values d n (- m (* q n)))))))

  (if (= n 0)
      1
      (* n (fact (- n 1)))))
(define (fact-cps n k)
  (if (= n 0)
      (k 1)
      (fact-cps (- n 1) (lambda (v) (k (* n v))))))

Fibonacci

(define (fib n)
  (if (< n 2)
      n
      (+ (fib (- n 1))
         (fib (- n 2)))))
(define (fib-iter n)
  (let iter ((a 0) (b 1) (c 0))
    (if (= c n)
        a
        (iter b (+ a b) (+ c 1)))))

Symbolic Differentiation

(define (diff exp var)
  (match exp
    (,num (guard (number? num)) 0)
    (,var^ (guard (symbol? var^)) (if (eq? var var^) 1 0))
    ((+ ,e1 ,e2) `(+ ,(diff e1 var) ,(diff e2 var)))
    ((* ,e1 ,e2) `(+ (* ,e1 ,(diff e2 var))
                     (* ,e2 ,(diff e1 var))))))

> (diff '(* (+ x a) (+ x b)) 'x)
(+ (* (+ x a) (+ 1 0)) (* (+ x b) (+ 1 0)))

Match

(define-syntax match
  (syntax-rules (guard)
    ((_ v) (error 'match "~s" v))
    ((_ v (pat (guard g ...) e ...) cs ...)
     (let ((fk (lambda () (match v cs ...))))
       (ppat v pat (if (and g ...) (let () e ...) (fk)) (fk))))
    ((_ v (pat e ...) cs ...)
     (let ((fk (lambda () (match v cs ...))))
       (ppat v pat (let () e ...) (fk))))))
(define-syntax ppat
  (syntax-rules (unquote)
    ((_ v () kt kf) (if (null? v) kt kf))
    ((_ v (unquote var) kt kf) (let ((var v)) kt))
    ((_ v (x . y) kt kf)
     (if (pair? v)
         (let ((vx (car v)) (vy (cdr v)))
           (ppat vx x (ppat vy y kt kf) kf))
         kf))
    ((_ v lit kt kf) (if (eqv? v (quote lit)) kt kf))))

Lisp-flavored XML

这个网页是通过Lisp-flavored XML编制的, 其句法如下:

<xml> ::= <string>
       |  (<symbol> (<attr>+) <xml>*)
       |  (<symbol> <xml>*)
<attr> ::= (<symbol> <string>)

将Lisp-flavored XML转换为XML的过程如下:

(define (Attr* attr*)
  (for-each
   (lambda (attr)
     (printf " ~s=~s" (car attr) (cadr attr)))
   attr*))
(define (Xml xml)
  (match xml
    (,str (guard (string? str)) (printf "~a" str))
    ((,sym ,attr* . ,xml*)
     (guard (pair? attr*) (pair? (car attr*)))
     (printf "<~s" sym)
     (Attr* attr*)
     (printf ">")
     (for-each Xml xml*)
     (printf "</~s>" sym))
    ((,sym . ,xml*)
     (printf "<~s>" sym)
     (for-each Xml xml*)
     (printf "</~s>" sym))))

And, Or, Not

and, or, not在概念上对应于合取, 析取, 否定. 然而, and和or遵循短路求值的原则, 因而并非过程.

   (and 1 2 3)
-> (and 2 3)
-> (and 3)
-> 3

   (or 1 2 3)
-> 1

如下定义的abs虽然是正确的, 但却是丑陋的, 不应该这么编写程序.

(define (abs x)
  (or (and (< x 0) (- x)) x))

递归定义在定义中直接或间接地引用自身.

(define (gcd a b)
  (if (= b 0)
      a

even?和odd?是互递归的过程, even?判断一个自然数是否是偶数, odd?判断一个自然数是否是奇数.

(define (even? n)
  (if (= n 0)
      #t

(define (odd? n)
  (if (= n 0)
      #f
      (even? (- n 1))))

(define (sqrt x)
  (define init 1.0)
  (define (square x)
    (* x x))
  (define tol 0.00001)
  (define (close-enough? a b)
    (< (abs (- a b)) tol))
  (define (average a b)
    (/ (+ a b) 2))
  (define (improve guess)
    (average guess (/ x guess)))
  (define (iter guess)
    (if (close-enough? (square guess) x)
        guess
        (iter (improve guess))))
  (iter init))

> (sqrt 2)
1.4142156862745097
> (sqrt 3)
1.7320508100147274
> (sqrt 5)
2.2360688956433634

Higher-order Procedures 高阶过程

高阶过程是以过程作为参数或返回值的过程.

(define (compose f g)
  (lambda (x)
    (f (g x))))

(define (repeated f n)
  (if (= n 0)
      (lambda (x) x)
      (compose f (repeated f (- n 1)))))

(define zero
  (lambda (f)
    (lambda (x) x)))
(define one
  (lambda (f)
    (lambda (x) (f x))))
(define two
  (lambda (f)
    (lambda (x) (f (f x)))))

(define (Sigma f a b)
  (if (> a b)
      0
      (+ (f a)
         (Sigma f (+ a 1) b))))

> (Sigma square 1 100)
338350
Recursive and Iterative Behaviors 递归和迭代行为

递归过程产生的计算并不总是具有相同的行为.

   (fact 3)
-> (* 3 (fact 2))
-> (* 3 (* 2 (fact 1)))
-> (* 3 (* 2 (* 1 (fact 0))))
-> (* 3 (* 2 (* 1 1)))
-> (* 3 (* 2 1))
-> (* 3 2)
-> 6

   (gcd 9 6)
-> (gcd 6 3)
-> (gcd 3 0)
-> 3

证明对于每个自然数n, (fact-cps n (lambda (v) v))等于(fact n).

(define (fact-cps n k)
  (if (= n 0)
      (k 1)
      (fact-cps (- n 1) (lambda (v) (k (* n v))))))

证明对于每个自然数n, (fib-cps n (lambda (v) v))等于(fib n).

(define (fib-cps n k)
  (if (< n 2)
      (k n)
      (fib-cps (- n 1)
               (lambda (v1)
                 (fib-cps (- n 2)
                          (lambda (v2)
                            (k (+ v1 v2))))))))

cps是continuation-passing style的缩写.

Complexity 复杂度

复杂度是计算所需的代价.

(define (expt x n)
  (if (= n 0)
      1
      (* x (expt x (- n 1)))))

首先证明对于每个整数x和每个自然数n, (expt^ x n)等于(expt x n); 接着分析expt和expt^的时间复杂度.

(define (expt^ x n)
  (let iter ((x x) (n n) (r 1))
    (cond ((= n 0) r)
          ((even? n)
           (iter (* x x) (/ n 2) r))
          (else
           (iter x (- n 1) (* x r))))))

首先证明对于每个自然数n, (fib-iter^ n)等于(fib n); 接着分析fib, fib-iter, fib-iter^的时间复杂度.

(define (fib-iter^ n)
  (let iter ((a 0) (b 1) (p 0) (q 1) (c n))
    (cond ((= c 0) a)
          ((even? c)
           (iter a
                 b
                 (+ (* p p) (* q q))
                 (+ (* 2 p q) (* q q))
                 (/ c 2)))
          (else
           (iter (+ (* p a) (* q b))
                 (+ (* p b) (* q a) (* q b))
                 p
                 q
                 (- c 1))))))

List Processing 列表处理

Scheme是LISP方言, 而LISP代表LISt Processor, 即列表处理器之意.

Pairs and Lists 序对和列表

cons构造一个序对, car和cdr分别代表第一投影和第二投影.

> (define p (cons 1 2))
> p
(1 . 2)
> (car p)
1
> (cdr p)
2

以下实现满足序对的公理.

(define (cons a d)
  (lambda (m)
    (m a d)))
(define (car p)
  (p (lambda (a d) a)))
(define (cdr p)
  (p (lambda (a d) d)))

空表是一个列表, 表达式(quote ())的值是一个空表, 但表达式()不是字面量.

如果lst是一个列表, 而x是一个任意的值, 那么(cons x lst)是一个列表.

> (quote ())
()
> (cons 1 (quote ()))
(1)
> (cons 1 (cons 2 (quote ())))
(1 2)
> (cons 1 (cons 2 (cons 3 (quote ()))))
(1 2 3)

list过程可以用来方便地构造列表.

> (list)
()
> (list 1)
(1)
> (list 1 2)
(1 2)
> (list 1 2 3)
(1 2 3)

quote是一种特殊形式, 它可以用来避免求值, 于是自然也可以用于构造列表, 这在列表的结构比较复杂时非常有用.

> (+ 1 (* 2 3))
7
> (quote (+ 1 (* 2 3)))
(+ 1 (* 2 3))

实际上此时我们引入了一种新的数据类型, 即符号. 符号和字符串具有诸多不同, 不应该混淆.

像(quote (+ 1 (* 2 3)))这样的表达式可以简写成'(+ 1 (* 2 3)). 尽管它看上去不符合Scheme一般的句法规则, 但读入的时候系统会将'(+ 1 (* 2 3))转换为(quote (+ 1 (* 2 3))), 因而不存在概念上的问题. 之后用到quote形式时, 一般均以简写形式出现.

'(+ 1 (* 2 3))足见quote形式构造列表的便利之处, 不然的话, 就要写成下面的样子.

> (cons '+ (cons 1 (cons (cons '* (cons 2 (cons 3 '()))) '())))
(+ 1 (* 2 3))
> (list '+ 1 (list '* 2 3))
(+ 1 (* 2 3))
Procedures on Lists 列表上的过程

本节是一些关于列表的基本过程.

length返回列表的长度.

(define (length lst)
  (if (null? lst)
      0
      (+ (length (cdr lst)) 1)))

> (length '())
0
> (length '(1))
1
> (length '((())))
1
> (length '(1 2 3))
3

list-ref从列表中取出指定位置的元素.

(define (list-ref lst n)
  (if (= n 0)
      (car lst)
      (list-ref (cdr lst) (- n 1))))

> (list-ref '(a b c) 0)
a
> (list-ref '(a b c) 1)
b
> (list-ref '(a b c) 2)
c

memq, memv, member判断一个元素是否在一个列表之中.

(define (memq x lst)
  (cond ((null? lst) #f)
        ((eq? x (car lst)) lst)
        (else (memq x (cdr lst)))))

> (memq 'y '(x y z))
(y z)
> (memq 'y '(x z))
#f
> (memq 'y '(x y y z))
(y y z)

memv和member分别使用eqv?和equal?比较相等.

assq, assv, assoc根据键对于关联列表进行索引.

(define (assq key alst)
  (cond ((null? alst) #f)
        ((eq? (caar alst) key) (car alst))
        (else (assq key (cdr alst)))))

> (assq 'y '((x 1) (y 2) (z 3)))
(y 2)
> (assq 'y '((x 1) (y 2) (y 3) (z 3)))
(y 2)
> (assq 'y '((x 1) (z 3)))
#f

类似地, assv和assoc分别使用eqv?和equal?比较相等.

A Toolbox for List Processing 列表处理工具箱

列表可以作为表示有限序列的媒介. 围绕这一媒介, 我们可以设计各种各样的工具. 它们很容易组合起来, 这正是列表处理的"表达力"所在.

map施行一个过程于列表的每个元素, 其结果仍是一个列表.

(define (map proc lst)
  (if (null? lst)
      '()
      (cons (proc (car lst))
            (map proc (cdr lst)))))

动笔计算一个实际的例子.

   (map square '(1 2 3))
-> (cons 1 (map square '(2 3)))
-> (cons 1 (cons 4 (map square '(3))))
-> (cons 1 (cons 4 (cons 9 (map square '()))))
-> (cons 1 (cons 4 (cons 9 '())))
-> (cons 1 (cons 4 '(9)))
-> (cons 1 '(4 9))
-> '(1 4 9)

filter从列表中过滤出满足谓词的元素.

(define (filter pred lst)
  (cond ((null? lst) '())
        ((pred (car lst))
         (cons (car lst) (filter pred (cdr lst))))
        (else (filter pred (cdr lst)))))

> (filter even? '(1 2 3 4 5 6))
(2 4 6)
> (filter odd? '(1 2 3 4 5 6))
(1 3 5)

append合并两个列表.

(define (append lst1 lst2)
  (if (null? lst1)
      lst2
      (cons (car lst1)
            (append (cdr lst1) lst2))))

根据append的定义可知, 第二个参数并非一定是列表.

> (append '(1 2 3) '(4 5 6))
(1 2 3 4 5 6)
> (append '(1 2 3) '())
(1 2 3)
> (append '() '(4 5 6))
(4 5 6)
> (append '(1 2 3) '4)
(1 2 3 . 4)

reverse反转一个列表.

以下写法是效率低下的.

(define (reverse lst)
  (if (null? lst)
      '()
      (append (reverse (cdr lst))
              (list (car lst)))))

这是更正确的写法.

(define (reverse lst)
  (let iter ((rest lst) (result '()))
    (if (null? rest)
        result
        (iter (cdr rest) (cons (car rest) result)))))

> (reverse '())
()
> (reverse '(1))
(1)
> (reverse '(1 2))
(2 1)
> (reverse '(1 2 3))
(3 2 1)

fold-left和fold-right是对于列表处理的抽象.

(define (fold-left proc init lst)
  (if (null? lst)
      init
      (fold-left proc (proc init (car lst)) (cdr lst))))
(define (fold-right proc init lst)
  (if (null? lst)
      init
      (proc (car lst)
            (fold-right proc init (cdr lst)))))

以下是呈现迭代行为的fold-right.

(define (fold-right proc init lst)
  (let iter ((rest (reverse lst)) (result init))
    (if (null? rest)
        result
        (iter (cdr rest) (proc (car rest) result)))))

以fold-left和fold-right表示map, filter, append, reverse.

(define (map proc lst)
  (fold-right (lambda (a d) (cons (proc a) d))
              '()
              lst))
(define (filter pred lst)
  (fold-right (lambda (a d) (if (pred a) (cons a d) d))
              '()
              lst))
(define (append lst1 lst2)
  (fold-right cons lst2 lst1))
(define (reverse lst)
  (fold-left (lambda (a d) (cons d a))
             '()
             lst))
More on List Processing 更多的列表处理
(define (append-map proc lst)
  (if (null? lst)
      '()
      (append (proc (car lst))
              (append-map proc (cdr lst)))))

> (append-map (lambda (x) (list x x)) '(1 2 3))
(1 1 2 2 3 3)

(define (product lst1 lst2)
  (append-map
   (lambda (y)
     (map (lambda (x) (list x y)) lst1))
   lst2))

(define (product . lst*)
  (if (null? lst*)
      '(())
      (append-map (lambda (d)
                    (map (lambda (a) (cons a d))
                         (car lst*)))
                  (apply product (cdr lst*)))))

Interpreters 解释器
(define (interp exp)
  (match exp
    (,int (guard (integer? int)) int)
    (,bool (guard (boolean? bool)) bool)
    ((if ,e1 ,e2 ,e3)
     (let ((v1 (interp e1)))
       (if (true? v1)
           (interp e2)
           (interp e3))))
    ((,op ,e1 ,e2)
     (guard (memq op '(+ - * =)))
     (let ((v1 (interp e1))
           (v2 (interp e2)))
       (case op
         ((+) (+ v1 v2))
         ((-) (- v1 v2))
         ((*) (* v1 v2))
         ((=) (= v1 v2)))))))
(define (true? val)
  (if (boolean? val)
      val
      (error 'interp "the predicate of if is not of type bool")))

Lexical Addressing 词法寻址
<exp> ::= <int>
       |  <bool>
       |  <var>
       |  (if <exp> <exp> <exp>)
       |  (lambda <exp>)
       |  (let <exp> <exp>)
       |  (<op> <exp> <exp>)
       |  (<exp> <exp>)
<var> ::= (var <int>)
<op> ::= + | - | * | =

(define (compil exp senv)
  (match exp
    (,int (guard (integer? int)) int)
    (,bool (guard (boolean? bool)) bool)
    (,var
     (guard (symbol? var))
     (let ((depth (apply-senv senv var)))
       `(var ,depth)))
    ((if ,e1 ,e2 ,e3)
     (let ((e1^ (compil e1 senv))
           (e2^ (compil e2 senv))
           (e3^ (compil e3 senv)))
       `(if ,e1^ ,e2^ ,e3^)))
    ((lambda ,x ,body)
     (let ((body^ (compil body (extend-senv x senv))))
       `(lambda ,body^)))
    ((let ,x ,e ,body)
     (let* ((e^ (compil e senv))
            (senv^ (extend-senv x senv))
            (body^ (compil body senv^)))
       `(let ,e^ ,body^)))
    ((,op ,e1 ,e2)
     (guard (memq op '(+ - * =)))
     (let ((e1^ (compil e1 senv))
           (e2^ (compil e2 senv)))
       `(,op ,e1^ ,e2^)))
    ((,rator ,rand)
     (let ((rator^ (compil rator senv))
           (rand^ (compil rand senv)))
       `(,rator^ ,rand^)))))

(define (empty-senv) '())
(define (extend-senv var senv)
  (cons var senv))
(define (apply-senv senv var)
  (let iter ((rest senv) (depth 0))
    (cond ((null? rest)
           (error 'apply-senv "unbound variable ~s" var))
          ((eq? (car rest) var) depth)
          (else (iter (cdr rest) (+ depth 1))))))

Types 类型

<type> ::= int
        |  bool
        |  (-> <type> <type>)
<exp> ::= <int>
       |  <bool>
       |  <var>
       |  (if <exp> <exp> <exp>)
       |  (lambda (<var> : <type>) <exp>)
       |  (let <var> <exp> <exp>)
       |  (<exp> <exp>)

+ : (-> int (-> int int))
- : (-> int (-> int int))
* : (-> int (-> int int))
zero? : (-> int bool)

(define (infer exp env)
  (match exp
    (,int (guard (integer? int)) 'int)
    (,bool (guard (boolean? bool)) 'bool)
    (,var (guard (symbol? var)) (apply-env env var))
    ((if ,e1 ,e2 ,e3)
     (let ((t1 (infer e1 env)))
       (unless (eq? t1 'bool)
         (error 'infer "the predicate of if is not of type bool"))
       (let ((t2 (infer e2 env))
             (t3 (infer e3 env)))
         (unless (equal? t2 t3)
           (error
            'infer
            "the consequent and alternative of if are not of the same type"))
         t2)))
    ((lambda (,x : ,t) ,body)
     (let* ((env^ (extend-env x t env))
            (t^ (infer body env^)))
       `(-> ,t ,t^)))
    ((let ,x ,e ,body)
     (let* ((t (infer e env))
            (env^ (extend-env x t env)))
       (infer body env^)))
    ((,rator ,rand)
     (let ((t1 (infer rator env))
           (t2 (infer rand env)))
       (match t1
         ((-> ,t3 ,t4)
          (unless (equal? t2 t3)
            (error 'infer "the rand is not of the right type"))
          t4)
         (,else
          (error 'infer "the rator is not of a function type")))))))

(define prim-env
  (extend-env
   '+ '(-> int (-> int int))
   (extend-env
    '- '(-> int (-> int int))
    (extend-env
     '* '(-> int (-> int int))
     (extend-env
      'zero? '(-> int bool)
      (empty-env))))))
(define (!- exp)
  (infer exp prim-env))
